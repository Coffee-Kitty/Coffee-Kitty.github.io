## [LCR 015. 找到字符串中所有字母异位词](https://leetcode.cn/problems/VabMRr/)



```python
class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
        """
            滑动窗口
        """
        hashmap = {}
        for tmp in p:
            hashmap[tmp] = hashmap.get(tmp,0)+1

        def check():
            for k,v in hashmap.items():
                if v!=0:
                    return False
            return True
        res = []
        i,j = -1,-1
        for j in range(len(s)):
            if s[j] in hashmap:
                hashmap[s[j]] -= 1

            if j < len(p)-1:
                continue
            
            if check():
                res.append(i+1)
            
            i += 1
            if s[i] in hashmap:
                hashmap[ s[i] ] += 1
        return res
                    

```

## [1004. 最大连续1的个数 III](https://leetcode.cn/problems/max-consecutive-ones-iii/description/)

```cpp
int longestOnes(vector<int>& nums, int k) {
    int res = 0;
    int l = 0, r = 0;
    int cnt = 0;
    for (r = 0; r < nums.size(); r++) {
        if (nums[r] == 0) {
            cnt++;
        }

        while (cnt > k) {
            if (nums[l] == 0) {
                cnt--;
            }
            l++;
        }

        res = max(res, r - l + 1);
    }
    return res;
}
```
