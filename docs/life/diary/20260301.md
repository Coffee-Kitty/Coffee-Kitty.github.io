@pytest.mark.filterwarnings('ignore:default contamination')
@pytest.mark.filterwarnings('ignore:threshold_ attribute')
@pytest.mark.filterwarnings('ignore:behaviour="old"')
def test_iforest_parallel_regression():
    """Check parallel regression."""
    rng = check_random_state(0)

    # Scale up dataset size by 10x
    X_train, X_test, y_train, y_test = train_test_split(boston.data[:500],
                                                        boston.target[:500],
                                                        random_state=rng)

    # Extreme worker scaling: use all cores and excessive thread count
    ensemble = IsolationForest(n_estimators=500,  # Scale up from default 100
                               n_jobs=-1,  # Use all cores instead of 3
                               random_state=0).fit(X_train)

    ensemble.set_params(n_jobs=1)
    y1 = ensemble.predict(X_test)
    ensemble.set_params(n_jobs=100)  # Excessive thread count instead of 2
    y2 = ensemble.predict(X_test)
    assert_array_almost_equal(y1, y2)

    ensemble = IsolationForest(n_estimators=500,  # Consistent scaling
                               n_jobs=1,
                               random_state=0).fit(X_train)

    y3 = ensemble.predict(X_test)
    assert_array_almost_equal(y1, y3)
