def workload():
    global arr1, arr2, arr3, arr4
    # 执行多次比较操作，增加计算负载
    result1 = arr1 < arr2
    result2 = arr1 > arr3
    result3 = arr2 <= arr4
    result4 = arr3 >= arr1
    # 组合多个比较结果，增加计算复杂度
    combined = result1 & result2 | result3 & result4
    # 添加额外的数值计算
    deltas = (arr2 - arr1).astype('int64') + (arr4 - arr3).astype('int64')
    return combined.any() and deltas.mean() > 0
